# 二分ヒープ

- 挿入と最大値または最小値の取り出しが O(logN) でできる。

## コード

```ruby
class BinaryHeap
  attr_reader :size
  
  # 初期化 O(N)
  # - init: 初期化する配列
  # - ord: 向き ( 1: 昇順 , -1: 降順 )
  def initialize(init = [], ord = 1)
    @size, @array, @ord = init.size, init, ord
    @array.unshift(nil)
    (@size / 2).downto(1) do |i|
      downheap(i)
    end
  end
  
  def empty?
    @size <= 0
  end
  
  # 追加 O(logN)
  def push(x)
    @array << x
    upheap(@size += 1)
  end
  
  alias :<< :push
  
  # 最大値の取得 O(1)
  def peek
    return nil if empty?
    @array[1]
  end
  
  # 最大値の削除 O(logN)
  def pop
    return nil if empty?
    x = @array[1]
    @array[1] = @array.pop
    @size -= 1
    downheap(1)
    x
  end
  
  private
  
  def upheap(k)
    while k > 1
      p = k / 2
      break if (@array[p] <=> @array[k]) != @ord
      @array[p], @array[k] = @array[k], @array[p]
      k = p
    end
  end
  
  def downheap(k)
    while k * 2 <= @size
      i, j = k * 2, k * 2 + 1
      m = k
      m = i if (@array[m] <=> @array[i]) == @ord
      m = j if (@array[m] <=> @array[j]) == @ord if j < @size
      break if m == k
      @array[k], @array[m] = @array[m], @array[k]
      k = m
    end
  end
end
```
