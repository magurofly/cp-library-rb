## 約数列挙 O(√n)

```ruby
def factors(n, &block)
  e = Enumerator.new do |y|
    k = 1
    while k * k < n
      if n % k == 0
        y << k
        y << n / k
      end
      k += 1
    end
    y << k if k * k == n
  end
  return e unless block_given?
  e.each(&block)
end
```

## 素数テーブル（線形篩） O(N)

```ruby
class PrimeTable
	def initialize(n)
		@lpf = [nil] * (n + 1)
		@primes = [2]
		(2 .. n).step(2) do |d| @lpf[d] = 2 end
		(3 .. n).step(2) do |d|
			unless @lpf[d]
				@lpf[d] = d
				@primes << d
			end
			@primes.each do |p|
				break if p * d > n or p > @lpf[d]
				@lpf[p * d] = p
			end
		end
	end
	def prime?(n); @lpf[n] == n; end
	def each(&block); @primes.each(&block); end
	def factorize(n); fs = Hash.new(0); while n > 1; fs[f = @lpf[n]] += 1; n /= f; end; fs; end
end
```

## 拡張ユークリッドの互除法 O(log N)

```ruby
def ext_gcd(a, b)
  s, t = b, a % b
  m0, m1 = 0, 1
  while t != 0
    u = s / t
    s -= t * u
    m0 -= m1 * u
    s, t = t, s
    m0, m1 = m1, m0
  end
  [s, m0]
end
```

## オイラーのφ関数 O(sqrt N)

- `require "prime"`

```ruby
def euler_phi(n)
  return n - 1 if n.prime?
  r = n
  i = 2
  while i * i <= n
    if n % i == 0
      r -= r / i
      n /= i while n % i == 0
    end
    i += 1
  end
  r -= r / n if n > 1
  r
end
```

## a^b^c mod m
- `euler_phi` を利用
- 参考: https://atcoder.jp/contests/abc228/submissions/27409810

```ruby
def pow3(a, b, c, m)
  phi_m = euler_phi(m)
  d, e, f = b, c, 1
  while e > 0
    f *= d if e.odd?
    return a.pow(b.pow(c, phi_m) + phi_m, m) if d >= phi_m / d
    d *= d
    e >>= 1
  end
  a.pow(f, m)
end
```
